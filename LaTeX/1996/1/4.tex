	Let $a_1=1$, $a_n=\frac{1}{n}\sum_{k=1}^{n-1}a_ka_{n-k}$ for $n\geq 2$. Show that
	\begin{enumerate}[label=(\roman*)]
		\item $\limsup_{n\to\infty}\lvert a_n\rvert^{1/n}<2^{-1/2}$;
		\item $\limsup_{n\to\infty}\lvert a_n\rvert^{1/n}\geq 2/3$.
	\end{enumerate}
